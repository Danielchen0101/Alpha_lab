async saveProviderConfig(config: AIProviderConfig): Promise<AIProviderConfigResponse> {
  try {
    const response = await api.post('/api/ai/provider/config', config);
    console.log('Save provider config response:', response.data);
    return response.data;
  } catch (error: any) {
    console.error('Save provider config error:', error);
    // 返回更详细的错误信息，而不是简单的 success: false
    return {
      success: false,
      config: config,
      error: error.response?.data?.error || error.message || '保存配置失败',
      message: '保存配置时发生错误，请检查网络连接和后端服务'
    };
  }
}

### TypeScript 编译错误修复
**错误1**: TS18046 - 'error' is of type 'unknown' (第323行)
- **根因**: `getAccountSnapshot` 方法中直接使用 `error.message`，但 TypeScript 要求对 `unknown` 类型进行安全收窄
- **修复**: 使用 `error instanceof Error ? error.message : String(error)` 安全提取错误消息

**错误2**: TS2322 - 返回对象包含 `error` 字段，但 `AIProviderConfigResponse` 类型没有这个字段 (第417行)
- **根因**: `saveProviderConfig` 方法在错误时返回包含 `error` 和 `message` 字段的对象，但 `AIProviderConfigResponse` 接口只定义 `success` 和 `config` 字段
- **修复**: 删除不符合接口定义的 `error` 和 `message` 字段，只返回 `{ success: false, config: config }`
